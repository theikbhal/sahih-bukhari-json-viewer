async function runAllBooks() {
  for (let i = 1; i <= 97; i++) {
    try {
      await scrapeHadiths(i);
    } catch (err) {
      console.error(`❌ Error scraping book ${i}:`, err.message);
    }
    await new Promise(resolve => setTimeout(resolve, 1500)); // Delay to avoid rate-limiting
  }
}

runAllBooks();
